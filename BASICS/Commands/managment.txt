

		Управление контейнерами


				(Ctrl + C) - Отключиться от контейнер и остановить его
				(Ctrl + P, Ctrl + Q) - отлючиться от контейнера без его остановки
				
				docker attach [OPTIONS] CONTAINER - наблюдение и взаимодействие за оснвоным процессом
				docker create - Создать контейнер из заданного образа, но не запускать его
				docker cp - копировать файлы между ФС контейнера и хоста
				docker exec - запускает заданную команду внутри контейнера
				
				docker kill - послылает команду основному процессу (ID=1) в контейнере. - Завешение работы.
				docker pause - временно приотсанавливает все процессы к контейнере.
				docker unpause - перезапуск контенейнера, остановленного pause.
				docker restart - перезапускает 1 или несколько контейнеров. (-t - интервал для отключения)

				docker rm - удлаяет один или несколько контейнеров
